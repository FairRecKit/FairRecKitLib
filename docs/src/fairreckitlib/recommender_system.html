 51    def __init__(self, data_dir: str, result_dir: str):
 52        """Construct the RecommenderSystem.
 53
 54        Initializes the data registry with available datasets on which the
 55        recommender system depends. It raises an IOError when the specified
 56        data directory does not exist. The result directory however is
 57        created when non-existing.
 58
 59        Args:
 60            data_dir: path to the directory that contains the datasets.
 61            result_dir: path to the directory to store computation results.
 62        """
 63        try:
 64            self.data_registry = DataRegistry(data_dir)
 65        except IOError as err:
 66            raise IOError('Failed to initialize DataRegistry: '
 67                          'unknown data directory => ' + data_dir) from err
 68
 69        self.result_dir = result_dir
 70        if not os.path.isdir(self.result_dir):
 71            os.mkdir(self.result_dir)
 72
 73        self.experiment_factory = create_experiment_factory()
 74        self.thread_processor = ThreadProcessor()
 75
 76    def abort_computation(self, computation_name: str) -> bool:
 77        """Attempt to abort a running computation thread.
 78
 79        The name of the computation is the same as specified in the configuration
 80        file when the computation is run. If the computation with the specified
 81        name does not exist this function returns False.
 82        Note that the computation is only requested to abort as soon as possible,
 83        therefore it might take a while until the computation actually stops.
 84
 85        Args:
 86            computation_name: name of the active computation thread to abort.
 87
 88        Returns:
 89            whether the computation is successfully requested to abort.
 90        """
 91        if not self.thread_processor.is_active_thread(computation_name):
 92            return False
 93
 94        self.thread_processor.stop(computation_name)
 95        return True

 97    def run_experiment(
 98            self,
 99            config: Union[PredictorExperimentConfig, RecommenderExperimentConfig],
100            *,
101            events: Dict[str, Callable[[Any], None]] = None,
102            num_threads: int = 0,
103            verbose: bool = True,
104            validate_config: bool = True) -> bool:
105        """Run an experiment with the specified configuration.
106
107        It is advised to validate the configuration (default) before running the
108        experiment, to make sure the configuration describes a valid experiment.
109        The configuration is invalid when it contains no selected datasets or
110        models, only the evaluation is optional. If the configuration is invalidated
111        this function will return False.
112
113        This functions raises an IOError when the result already exists or a TypeError
114        when the provided configuration is not a valid experiment configuration.
115        Lastly a KeyError is raised when a computation with the same name is already active.
116
117        Args:
118            events: the external events to dispatch during the experiment.
119            config: the configuration of the experiment.
120            num_threads: the max number of threads the experiment can use.
121            verbose: whether the internal events should give verbose output.
122            validate_config: whether to validate the configuration beforehand.
123
124        Returns:
125            whether the experiment successfully started.
126        """
127        result_dir = os.path.join(self.result_dir, config.name)
128        if os.path.isdir(result_dir):
129            raise IOError('Result already exists: ' + result_dir)
130
131        if not isinstance(config, (PredictorExperimentConfig, RecommenderExperimentConfig)):
132            raise TypeError('Invalid experiment configuration type.')
133
134        if validate_config:
135            parser = Parser(verbose)
136            config = parser.parse_experiment_config(config.to_yml_format(),
137                                                    self.data_registry,
138                                                    self.experiment_factory)
139            if config is None:
140                return False
141
142        self.thread_processor.start(ThreadExperiment(
143            config.name,
144            events,
145            verbose,
146            pipeline_config=ExperimentPipelineConfig(
147                result_dir,
148                self.data_registry,
149                self.experiment_factory,
150                config,
151                0,
152                1,
153                num_threads
154            )
155        ))
156
157        return True

159    def run_experiment_from_yml(
160            self,
161            file_path: str,
162            *,
163            events: Dict[str, Callable[[Any], None]] = None,
164            num_threads: int = 0,
165            verbose: bool = True) -> bool:
166        """Run an experiment from a yml file.
167
168        The configuration in the file is validated before starting the experiment.
169        It is invalid when it contains no selected datasets or models,
170        only the evaluation is optional.  If the configuration is invalidated
171        this function will return False.
172
173        This function raises a FileNotFoundError when the specified yml file
174        is not found or an IOError when the result already exists. Lastly a KeyError
175        is raised when a computation with the same name is already active.
176
177        Args:
178            events: the external events to dispatch during the experiment.
179            file_path: path to the yml file without extension.
180            num_threads: the max number of threads the experiment can use.
181            verbose: whether the internal events should give verbose output.
182
183        Returns:
184            whether the experiment successfully started.
185        """
186        try:
187            parser = Parser(verbose)
188            config = parser.parse_experiment_config_from_yml(file_path,
189                                                             self.data_registry,
190                                                             self.experiment_factory)
191            if config is None:
192                return False
193        except FileNotFoundError as err:
194            raise FileNotFoundError(errno.ENOENT, 'Config file not found', file_path) from err
195
196        return self.run_experiment(
197            config,
198            events=events,
199            num_threads=num_threads,
200            verbose=verbose,
201            validate_config=False
202        )

204    def validate_experiment(
205            self,
206            result_dir: str,
207            num_runs: int,
208            *,
209            events: Dict[str, Callable[[Any], None]] = None,
210            num_threads: int = 0,
211            verbose: bool = True) -> bool:
212        """Validate an experiment for an additional number of runs.
213
214        The configuration file is expected to be stored in the specified result directory.
215        Moreover, the configuration is validated before starting the experiment validation.
216        If the configuration is invalidated this function will return False.
217
218        This functions raises an IOError when the result already exists or
219        a FileNotFoundError when the configuration file was not found in the
220        specified result directory. Lastly a KeyError is raised when a computation
221        with the same name is already active, meaning it is not possible to validate
222        an active experiment computation until it is done.
223
224        Args:
225            events: the external events to dispatch during the experiment.
226            result_dir: path to an existing experiment result directory.
227            num_runs: the number of runs to validate the experiment.
228            num_threads: the max number of threads the experiment can use.
229            verbose: whether the internal events should give verbose output.
230
231        Returns:
232            whether the experiment successfully started.
233        """
234        result_dir = os.path.join(self.result_dir, result_dir)
235        if not os.path.isdir(result_dir):
236            raise IOError('Result does not exist: ' + result_dir)
237
238        config_path = os.path.join(result_dir, 'config')
239        try:
240            parser = Parser(verbose)
241            config = parser.parse_experiment_config_from_yml(config_path,
242                                                             self.data_registry,
243                                                             self.experiment_factory)
244            if config is None:
245                return False
246        except FileNotFoundError as err:
247            raise FileNotFoundError(errno.ENOENT, 'Config file not found', config_path) from err
248
249        self.thread_processor.start(ThreadExperiment(
250            config.name,
251            events,
252            verbose,
253            pipeline_config=ExperimentPipelineConfig(
254                result_dir,
255                self.data_registry,
256                self.experiment_factory,
257                config,
258                resolve_experiment_start_run(result_dir),
259                num_runs,
260                num_threads
261            )
262        ))
263
264        return True
