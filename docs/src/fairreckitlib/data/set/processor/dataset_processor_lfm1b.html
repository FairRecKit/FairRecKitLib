122    def load_artist_genres_allmusic(self) -> Optional[pd.DataFrame]:
123        """Load the artist allmusic genres file.
124
125        Returns:
126            the loaded artist name/genre table or None on failure.
127        """
128        try:
129            genres = pd.read_csv(
130                os.path.join(self.dataset_dir, 'LFM-1b_artist_genres_allmusic.txt'),
131                sep='\t',
132                names=['artist_name'] + [str(i) for i in range(0, len(ALL_MUSIC_GENRES))]
133            )
134        except FileNotFoundError:
135            return None
136
137        # remove duplicate rows where artist name is the same
138        genres.drop_duplicates(subset='artist_name', inplace=True)
139        # extract and drop artist name column
140        artist_genres = pd.DataFrame(genres['artist_name'])
141        genres.drop('artist_name', inplace=True, axis=1)
142
143        # map allmusic genre id to genre name
144        for col in genres:
145            genres[col] = genres[col].map(lambda i: ALL_MUSIC_GENRES[int(i)], na_action='ignore')
146
147        # add genres column
148        artist_genres['artist_genres'] = genres.apply(lambda x: x.str.cat(sep='|'), axis=1)
149
150        return artist_genres

172    def process_artist_table(self) -> Optional[DatasetTableConfig]:
173        """Process the artist table.
174
175        Extends the table with artist gender and genres information when available.
176
177        Returns:
178            the artist table configuration or None on failure.
179        """
180        artist_table_config = create_dataset_table_config(
181            'LFM-1b_artists.txt',
182            ['artist_id'],
183            ['artist_name']
184        )
185
186        try:
187            artist_table = artist_table_config.read_table(self.dataset_dir)
188        except FileNotFoundError:
189            artist_table = pd.DataFrame()
190            artist_table_config.columns.pop()
191
192        # add artist gender when available
193        gender_table = self.load_artist_gender_json()
194        if gender_table is not None:
195            # replace artist table when missing
196            if len(artist_table) == 0:
197                artist_table = gender_table
198            # merge artist table with gender
199            else:
200                artist_table = pd.merge(artist_table, gender_table, how='left', on='artist_id')
201            artist_table_config.columns += ['artist_gender']
202
203        # no need to continue if the previous failed
204        if len(artist_table) == 0:
205            return None
206
207        if 'artist_name' in artist_table_config.columns:
208            # attempt to load artist name / genre table
209            artist_genres = self.load_artist_genres_allmusic()
210            if artist_genres is not None:
211                # merge artist table with genres
212                artist_table = pd.merge(artist_table, artist_genres, how='left', on='artist_name')
213                artist_table_config.columns += ['artist_genres']
214
215        artist_table_config.file.name = TABLE_FILE_PREFIX + self.dataset_name + '_artists.tsv.bz2'
216        artist_table_config.file.compression = 'bz2'
217        artist_table_config.num_records = len(artist_table)
218
219        # store generated artist table
220        artist_table_config.save_table(artist_table, self.dataset_dir)
221
222        return artist_table_config

263    def process_user_artist_matrix(self) -> Optional[DatasetMatrixConfig]:
264        """Process the user-artist-count matrix.
265
266        The user-item matrix is stored in a matlab file in CSR compatible format,
267        together with the user and item indices. The matrix is converted
268        to a dataframe and the indices for the indirection arrays are flattened.
269
270        Returns:
271            the matrix configuration or None on failure.
272        """
273        try:
274            mat_file = os.path.join(self.dataset_dir, 'LFM-1b_LEs.mat')
275            # load matrix as described in the paper
276            csr_matrix, idx_users, idx_artists = _load_lfm_1b_mat(mat_file)
277        except FileNotFoundError:
278            return None
279
280        matrix_name = 'user-artist-count'
281
282        # create and save user indirection array
283        user_list = list(map(lambda i: i[0], idx_users))
284        user_index_config = DatasetIndexConfig(
285            TABLE_FILE_PREFIX + self.dataset_name + '_' + matrix_name + '_user_indices.hdf5',
286            'user_id',
287            len(user_list)
288        )
289        user_index_config.save_indices(self.dataset_dir, user_list)
290
291        # create and save artist indirection array
292        artist_list = list(map(lambda i: i[0], idx_artists))
293        artist_index_config = DatasetIndexConfig(
294            TABLE_FILE_PREFIX + self.dataset_name + '_' + matrix_name + '_item_indices.hdf5',
295            'artist_id',
296            len(artist_list)
297        )
298        artist_index_config.save_indices(self.dataset_dir, artist_list)
299
300        # convert csr to dataframe
301        coo_matrix = pd.DataFrame.sparse.from_spmatrix(csr_matrix).sparse.to_coo()
302        user_artist_matrix = pd.DataFrame()
303        user_artist_matrix['user_id'] = coo_matrix.row
304        user_artist_matrix['artist_id'] = coo_matrix.col
305        user_artist_matrix['matrix_count'] = coo_matrix.data
306
307        # create matrix table configuration
308        user_artist_table_config = create_dataset_table_config(
309            TABLE_FILE_PREFIX + self.dataset_name + '_' + matrix_name + '_matrix.tsv.bz2',
310            ['user_id', 'artist_id'],
311            ['matrix_count'],
312            compression='bz2',
313            foreign_keys=['user_id', 'artist_id'],
314            num_records=len(user_artist_matrix)
315        )
316
317        # store the resulting matrix
318        user_artist_table_config.save_table(user_artist_matrix, self.dataset_dir)
319
320        return DatasetMatrixConfig(
321            user_artist_table_config,
322            user_artist_matrix['matrix_count'].min(),
323            user_artist_matrix['matrix_count'].max(),
324            DATASET_RATINGS_IMPLICIT,
325            user_index_config,
326            artist_index_config
327        )

415def _load_lfm_1b_mat(file_path: str) -> Tuple[sparse.csr_matrix, np.array, np.array]:
416    """Load the LFM-1B dataset from the matlab file.
417
418    Args:
419        file_path: the path to the matlab file.
420
421    Returns:
422        the matrix and user / artist indirection arrays.
423    """
424    with h5py.File(file_path, 'r') as mat_file:
425        csr_matrix = sparse.csr_matrix((
426            mat_file['/LEs/']["data"],
427            mat_file['/LEs/']["ir"],
428            mat_file['/LEs/']["jc"]
429        )).transpose()
430        idx_users = np.array(mat_file.get('idx_users')).astype(np.int64)
431        idx_artists = np.array(mat_file.get('idx_artists')).astype(np.int64)
432        return csr_matrix, idx_users, idx_artists

def load_artist_genres_allmusic(self) -> Optional[pandas.core.frame.DataFrame]: 
123    def load_artist_genres_allmusic(self) -> Optional[pd.DataFrame]:
124        """Load the artist allmusic genres file.
125
126        Returns:
127            the loaded artist name/genre table or None on failure.
128        """
129        try:
130            genres = pd.read_csv(
131                os.path.join(self.dataset_dir, 'LFM-1b_artist_genres_allmusic.txt'),
132                sep='\t',
133                names=['artist_name'] + [str(i) for i in range(0, len(ALL_MUSIC_GENRES))]
134            )
135        except FileNotFoundError:
136            return None
137
138        # remove duplicate rows where artist name is the same
139        genres.drop_duplicates(subset='artist_name', inplace=True)
140        # extract and drop artist name column
141        artist_genres = pd.DataFrame(genres['artist_name'])
142        genres.drop('artist_name', inplace=True, axis=1)
143
144        # map allmusic genre id to genre name
145        for col in genres:
146            genres[col] = genres[col].map(lambda i: ALL_MUSIC_GENRES[int(i)], na_action='ignore')
147
148        # add genres column
149        artist_genres['artist_genres'] = genres.apply(lambda x: x.str.cat(sep='|'), axis=1)
150
151        return artist_genres

173    def process_artist_table(self) -> Optional[DatasetTableConfig]:
174        """Process the artist table.
175
176        Extends the table with artist gender and genres information when available.
177
178        Returns:
179            the artist table configuration or None on failure.
180        """
181        artist_table_config = create_dataset_table_config(
182            'LFM-1b_artists.txt',

184            ['artist_name']
185        )
186
187        try:
188            artist_table = artist_table_config.read_table(self.dataset_dir)

192
193        # add artist gender when available

195        if gender_table is not None:

203
204        # no need to continue if the previous failed
205        if len(artist_table) == 0:
206            return None
207
208        if 'artist_name' in artist_table_config.columns:
209            # attempt to load artist name / genre table
210            artist_genres = self.load_artist_genres_allmusic()
211            if artist_genres is not None:
212                # merge artist table with genres
213                artist_table = pd.merge(artist_table, artist_genres, how='left', on='artist_name')
214                artist_table_config.columns += ['artist_genres']
215
216        artist_table_config.file.name = TABLE_FILE_PREFIX + self.dataset_name + '_artists.tsv.bz2'
217        artist_table_config.file.compression = 'bz2'
218        artist_table_config.num_records = len(artist_table)
